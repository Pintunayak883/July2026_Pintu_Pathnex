## 1️⃣ Terraform — Create EC2 (any instance type)

provider "aws" {
    region = "eu-west-3"
}

resource "aws_vpc" "my_vpc" {
    cidr_block = "10.0.0.16"

    tags = {
        Name = "My-VPC"
    }
}

resource "aws_subnet" "my_subnet" {
    vcp_id                  = aws_vpc.my_vpc.id
    cidr_block              = "10.0.1.0/24"
    availability_zone       = "eu-west-3a"
    map_public_ip_on_launch = true

    tags = {
        Name = "Public-Subnet
    }
}

resource "aws_route_table" "my_rt" {
    vpc_id = aws_vpc.my_vpc.id
    
    route {
        cidr_block  = "0.0.0.0/0"
        gateway_id  = aws_internet_gateway.my_igw.id
    }

    tags = {
        Name = "Public-RouteTable"
    }
}

resource "aws_security_group" "my_sg" {
    name      = "allow-ssh-http"
    vpc_id    = aws_vpc.my_vpc.id

    ingress {
        description  = "SSH"
        form_port    = 22
        to_port      = 22
        protocol     = "tcp"
        cidr_block   = ["0.0.0.0/0"]
    }

    ingress {
        description = "HTTP"
        from_port   = 80
        to_port     = 80
        protocol    = "tcp"
        cidr_block  = ["0.0.0.0/0"]
    }

    egress {
        from_port   = 0
        to_port     = 0
        protocol    = "-1"
        cidr_block  = ["0.0.0.0/0"]
    }

    tags = {
        Name = "My-SecurityGroup
    }
}

resource "aws_instance" "my_ec2" {
    ami                    = "ami-0abcd1234abcd1234"
    instance_type          = "c5.xlarge"
    subnet_id              = aws_subnet.my_subnet.id
    vpc_security_group_ids = [aws_security_group.my_sg.id]
    key_name               = "your-keypair-name"

    tags = {
        Name  = "Terraform-EC2"
    } 
}

## 2️⃣ Ansible — Install Nginx & Deploy HTML File
--- 
- hosts: web
  become: yes

  tasks:
    - name: Install Nginx 
      yum:
        name: nginx
        state: present
        update_cache:yes

    - name: Copy html
      copy:
        src: index.html
        dest: /var/www/html/index.html

    - name: Start nginx
      service:
         name: nginx
         state:started
         enabled: yes

## 3️⃣ Kubernetes — Create Deployment + Service

apiVersion: apps/v1
kind: Deployment

metadata:
    name: nginx-Deployment

spec:
  replicas: 2

  selector:
     matchLabels:
        app: nginx
  
  template:
     metadata:
       labels:
         app: nginx

     spec:
       containers:
         
         - name: nginx
           image: nginx
           ports:
            - containerPort: 80

---
apiVersion: v1
kind: service

metadata:
   name: nginx-service

 spec:
   selector:
     app: nginx

    ports:

    - port: 80

     targetPort: 80

    type: NodePort

## Ansible - setup Nginx with Reverse Proxy

```yaml
- name: Setup Nginx with Reverse Proxy
  hosts: all
  become: yes
  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present
    
    - name: Configure nginx as reverse Proxy
      template:
        src: reverse-proxy-conf.j2
        dest: /etc/nginx/conf.d/reverse-proxy.conf
    - name: Restart nginx service
      service:
        name: nginx
        state: restarted
        enabled: yes

# Kubernetes - Ingress with Nginx Controller
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
   name: pathnex-ingress
   annotations:
      nginx.ingress.kubernetes.io/rewrite-target: /
   spec:
     rules:
       - host: pathnex.local
         http:
           paths:
           - path: /
             pathType: Prefix
             backend:
               service:
                  name: pathnex-service
                  port:
                    number: 80
